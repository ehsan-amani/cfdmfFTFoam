73 inline void Foam::bubbleData::updateBubbleVolume()
74 {
75  if (!elPointsFlag) pointToElementMapping();
76 
77  volume = 0;
78  scalar curTetrahedralVolume = 0;
79  vector elementCenter = vector::zero;
80 
81  forAll(eL, elementI)
82  {
83  elementInfo& elD = eL[elementI];
84 
85  vector e01 = ( elD.points[1] - elD.points[0]);
86  vector e02 = ( elD.points[2] - elD.points[0]);
87  vector Sf = 0.5 * (e01 ^ e02);
88 
89  elementCenter = ( elD.points[0] + elD.points[1] + elD.points[2])/3.0;
90 
91  curTetrahedralVolume = ( elementCenter & Sf )/3.0;
92 
93  volume = volume + curTetrahedralVolume;
94  }
95 
96  volume = mag(volume);
97 
98  volumeFlag = true;
99 }
